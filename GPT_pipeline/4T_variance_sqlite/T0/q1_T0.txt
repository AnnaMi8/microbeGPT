SELECT AVG(count) 
FROM abundance AS a
JOIN HealthyStoolSamples AS m ON a.sample_id = m.sample_id
JOIN taxonomy AS t ON a.species = t.species
WHERE t.genus = 'Faecalibacterium'
  AND t.species = 'Faecalibacterium prausnitzii';
