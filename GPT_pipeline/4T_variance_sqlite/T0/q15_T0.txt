1
SELECT t.phylum, SUM(a.count) AS total_abundance
FROM taxonomy AS t
JOIN abundance AS a ON t.species = a.species
JOIN HealthyStoolSamples AS m ON a.sample_id = m.sample_id
GROUP BY t.phylum
ORDER BY total_abundance DESC
LIMIT 1;
