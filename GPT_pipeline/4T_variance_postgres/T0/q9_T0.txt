1
SELECT species, SUM(count) AS total_abundance
FROM ds_mgpt_mgpt.abundance AS a
JOIN ds_mgpt_mgpt.metadata AS m ON a.sample_id = m.sample_id
WHERE m.disease != 'healthy'
GROUP BY species
ORDER BY total_abundance DESC
LIMIT 10;
